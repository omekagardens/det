import numpy as np
import matplotlib.pyplot as plt

# --- DET 3.0 Constants ---
GLOBAL_EVENTS = 800
DT = 0.1

class Agent:
    def __init__(self, name, alignment_strategy):
        self.name = name
        self.strategy = alignment_strategy # 'Believer' or 'Skeptic'
        self.F = 0.0      # Mass/History
        self.chi = 0.0    # Drag/Entropy
        self.psi = 1.0    # Connection to Source
        self.P = 1.0      # Presence (Time Speed)
        
    def update(self, epoch, reservoir_limit):
        # --- PHYSICS EVOLUTION ---
        
        # 1. PHASE 1: GARDEN (No Mass, Pure Flow)
        if epoch == 'GARDEN':
            self.F = 0.1 # Very light
            self.chi = 0.0
            self.psi = 1.0
            
        # 2. PHASE 2: FALL (Accumulation of Burden)
        elif epoch == 'FALL':
            self.F += 0.5 * DT   # Accumulating History/Karma
            self.chi += 0.2 * DT # Entropy builds up
            # Reservoir is THROTTLED (Law) - can't clear drag fast enough
            reservoir_limit = 2.0 
            self.psi = 0.5 # Connection degrades under stress
            
        # 3. PHASE 3: GRACE (The Split)
        elif epoch == 'GRACE':
            self.F += 0.1 * DT # History still grows
            # Reservoir is INFINITE now
            reservoir_limit = 9999.0
            
            if self.strategy == 'Believer':
                self.psi = 1.0 # Re-align
            else:
                self.psi = 0.1 # Stay in old physics
                self.chi += 0.1 * DT # Continue accumulating drag
                
        # 4. PHASE 4: UTOPIA (Crystallized Scaffolding)
        elif epoch == 'UTOPIA':
            # Mass is Fixed (The Scaffolding is complete)
            # F stops growing
            
            # Everyone eventually aligns (Evolution)
            self.psi = 1.0
            reservoir_limit = 9999.0

        # --- UNIVERSAL CALCULATIONS ---
        
        # Flux J = Psi * (Limit - Chi)
        flux = self.psi * (reservoir_limit - self.chi) * 0.1 * DT
        # Apply Flux (Cleaning)
        self.chi = max(0, self.chi - flux)
        
        # Calculate Presence P = 1 / (1 + Mass + Drag)
        # Note: In Utopia, Mass is decoupled (Scaffolding Physics), 
        # so we effectively remove F from the denominator if Psi=1 and Chi=0
        
        mass_effect = self.F
        if epoch == 'UTOPIA':
            mass_effect = 0 # Decoupled!
            
        self.P = 1.0 / (1.0 + (0.1 * mass_effect) + self.chi)
        
        return self.P

# --- Run Simulation ---
believer = Agent("Aligned", "Believer")
skeptic = Agent("Hoarder", "Skeptic")

epochs = []

    limit = 0
    
    if k < 100:

    elif k < 300:
        epoch = 'FALL'

        epoch = 'GRACE'
    else:
        epoch = 'UTOPIA'
        
    p_b = believer.update(epoch, limit)
    p_s = skeptic.update(epoch, limit)
    
    history_believer.append(p_b)
    history_skeptic.append(p_s)

# --- Visualization ---
fig, ax = plt.subplots(figsize=(10, 6))

# Plot lines
ax.plot(history_believer, color='gold', linewidth=3, label="Aligned Agent (The Church)")
ax.plot(history_skeptic, color='grey', linestyle='--', label="Misaligned Agent (The World)")

# Draw Epoch Boundaries
ax.axvline(x=100, color='black', linestyle=':')
ax.axvline(x=300, color='green', linewidth=2, linestyle='--') # The Cross
ax.axvline(x=600, color='blue', linestyle=':')

# Annotations
ax.text(10, 0.5, "PHASE 1:\nGARDEN", fontsize=10)

ax.set_title("The Evolution of Physics: DET 3.0 Cosmology")
ax.set_ylabel("Quality of Existence (Presence P)")
ax.set_xlabel("Time (Events)")
ax.legend(loc='lower left')
ax.grid(True)

plt.tight_layout()
plt.show()
