### 6.1 Modeling Body-Spirit Oscillation

```python
class OscillatingAgent:
    """
    Agent that naturally oscillates between body and spirit dominance.
    Models the common human experience.
    """
    def __init__(self):
        self.mode_weight = 0.5  # 0=full spirit, 1=full body
        self.q = 0.1  # Initial debt
        self.C = 0.5  # Initial coherence
        self.F = 1.0  # Initial resource
        self.fear = 0.0

    def step(self, environment):
        # Environment can trigger fear
        threat = environment.get_threat_level()
        self.fear = 0.9 * self.fear + 0.1 * threat

        # Fear pulls toward body-mode
        fear_pull = self.fear * 0.3

        # High coherence pulls toward spirit-mode
        coherence_pull = self.C * 0.2

        # High q (debt ceiling) limits spirit-mode access
        debt_limit = self.q * 0.1

        # Update mode weight
        self.mode_weight = np.clip(
            self.mode_weight + fear_pull - coherence_pull + debt_limit,
            0, 1
        )

        # Make decision based on current mode
        if self.mode_weight < 0.5:
            decision = self._spirit_decision(environment)
        else:
            decision = self._body_decision(environment)

        # Decision affects q accumulation
        if decision == 'coherent':
            self.q += 0.001  # Small identity-q
            self.C += 0.01   # Coherence grows
        else:
            self.q += 0.005  # More damage-q
            self.C -= 0.01   # Coherence degrades

        return decision

    def get_state(self):
        return {
            'mode': 'spirit' if self.mode_weight < 0.5 else 'body',
            'mode_weight': self.mode_weight,
            'q': self.q,
            'C': self.C,
            'fear': self.fear
        }
```
